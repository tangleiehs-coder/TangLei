## 修正方案

### 方案A: 单流生成（推荐，无累积误差）

不再分段拼接，改用单流方式：

```python
# 1. 生成所有帧图片序列
for i, (start, end) in enumerate(时间段列表):
    duration = end - start
    frames = int(duration * 30)  # 30fps
    for f in range(frames):
        # 复制当前段的画面帧
        shutil.copy(f"full_{i:02d}.png", f"frames/frame_{总帧数:06d}.png")
        总帧数 += 1

# 2. 一次性编码
ffmpeg -framerate 30 -i frames/frame_%06d.png -i 完整音频.mp3 -c:v libx264 -c:a aac 输出.mp4
```


## 自动化核对脚本

```python
import subprocess
import json

def verify_sync(video_path, audio_path, check_points):
    """
    核对关键时间点音画同步
    check_points: [(视频秒数, 期望画面描述), ...]
    """
    # 生成带时间戳的截图用于验证
    for ts, desc in check_points:
        subprocess.run([
            "ffmpeg", "-y", "-i", video_path,
            "-ss", str(ts), "-vframes", "1",
            f"verify_{ts}.png"
        ])
        print(f"时间点 {ts}s 截图已生成，期望: {desc}")
    
    # 获取时长对比
    video_dur = float(subprocess.check_output([
        "ffprobe", "-v", "error", "-show_entries", "format=duration",
        "-of", "default=noprint_wrappers=1:nokey=1", video_path
    ]))
    audio_dur = float(subprocess.check_output([
        "ffprobe", "-v", "error", "-show_entries", "format=duration",
        "-of", "default=noprint_wrappers=1:nokey=1", audio_path
    ]))
    
    diff = abs(video_dur - audio_dur)
    status = "✅ 合格" if diff < 1 else "⚠️ 可接受" if diff < 3 else "❌ 不合格"
    
    return {
        "video_duration": video_dur,
        "audio_duration": audio_dur,
        "diff_seconds": diff,
        "status": status
    }
```
